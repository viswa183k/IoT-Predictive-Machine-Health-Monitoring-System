/*
  IoT Predictive Machine Health Monitoring System
  ESP32 + MPU6050 + DHT22 + ThingSpeak
*/

#include <WiFi.h>
#include <HTTPClient.h>
#include <Wire.h>
#include <MPU6050.h>
#include "DHT.h"

#define DHTPIN 4
#define DHTTYPE DHT22
#define ALERT_LED 2

const char* ssid = "YOUR_WIFI_NAME";
const char* password = "YOUR_WIFI_PASSWORD";

String apiKey = "YOUR_THINGSPEAK_API_KEY";
String server = "http://api.thingspeak.com/update";

MPU6050 mpu;
DHT dht(DHTPIN, DHTTYPE);

unsigned long operatingHours = 0;

void setup() {

  Serial.begin(115200);

  Wire.begin();
  mpu.initialize();
  dht.begin();

  pinMode(ALERT_LED, OUTPUT);

  WiFi.begin(ssid, password);

  while (WiFi.status() != WL_CONNECTED) {
    delay(500);
    Serial.print(".");
  }

  Serial.println("\nWiFi Connected");
}

void loop() {

  int16_t ax, ay, az;

  mpu.getAcceleration(&ax, &ay, &az);

  float vibration =
      sqrt((long)ax * ax +
           (long)ay * ay +
           (long)az * az);

  float temperature = dht.readTemperature();

  operatingHours++;

  bool abnormal =
      vibration > 22000 ||
      temperature > 65 ||
      operatingHours > 50000;

  digitalWrite(ALERT_LED, abnormal);

  Serial.print("Temp: ");
  Serial.print(temperature);
  Serial.print(" C  Vibration: ");
  Serial.print(vibration);
  Serial.print("  Hours: ");
  Serial.println(operatingHours);

  if (WiFi.status() == WL_CONNECTED) {

    HTTPClient http;

    String url =
      server +
      "?api_key=" + apiKey +
      "&field1=" + String(temperature) +
      "&field2=" + String(vibration) +
      "&field3=" + String(operatingHours);

    http.begin(url);
    http.GET();
    http.end();
  }

  delay(15000);
}
